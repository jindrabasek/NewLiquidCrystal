00081                          uint8_t d0, uint8_t d1, uint8_t d2, uint8_t d3,
00082                          uint8_t d4, uint8_t d5, uint8_t d6, uint8_t d7)

00086    // Initialize the IO pins
00087    // -----------------------
00088    
00089    _rs_pin = rs;

00091    _enable_pin = enable;
00092    
00093    _data_pins[0] = d0;
00094    _data_pins[1] = d1;
00095    _data_pins[2] = d2;
00096    _data_pins[3] = d3; 
00097    _data_pins[4] = d4;
00098    _data_pins[5] = d5;
00099    _data_pins[6] = d6;
00100    _data_pins[7] = d7;
00101    
00102    // Initialize the IO port direction to OUTPUT
00103    // ------------------------------------------
00104    
00105    for ( i = 0; i < 4; i++ )
00106    {
00107       pinMode ( _data_pins[i], OUTPUT );
00108    }
00109    
00110    // Initialize the rest of the ports if it is an 8bit controlled LCD
00111    // ----------------------------------------------------------------
00112    
00113    if ( !fourbitmode )
00114    {
00115       for ( i = 4; i < 8; i++ )
00116       {
00117          pinMode ( _data_pins[i], OUTPUT );
00118       }
00119    }
00120    pinMode(_rs_pin, OUTPUT);
00121    
00122    // we can save 1 pin by not using RW. Indicate by passing 255 instead of pin#
00123    if (_rw_pin != 255) 
00124    { 
00125       pinMode(_rw_pin, OUTPUT);
00126    }
00127    
00128    pinMode(_enable_pin, OUTPUT);
00129    
00130    // Initialise displaymode functions to defaults: LCD_1LINE and LCD_5x8DOTS
00131    // -------------------------------------------------------------------------
00132    if (fourbitmode)
00133       _displayfunction = LCD_4BITMODE | LCD_1LINE | LCD_5x8DOTS;
00134    else 
00135       _displayfunction = LCD_8BITMODE | LCD_1LINE | LCD_5x8DOTS;
00136    
00137    // Now we pull both RS and R/W low to begin commands
00138    digitalWrite(_rs_pin, LOW);
00139    digitalWrite(_enable_pin, LOW);
00140    
00141    if (_rw_pin != 255) 
00142    { 
00143       digitalWrite(_rw_pin, LOW);
00144    }
00145 }

00153 void LiquidCrystal::send(uint8_t value, uint8_t mode) 
00154 {
00155    digitalWrite( _rs_pin, mode );
00156    
00157    // if there is a RW pin indicated, set it low to Write
00158    // ---------------------------------------------------
00159    if (_rw_pin != 255) 
00160    { 
00161       digitalWrite(_rw_pin, LOW);
00162    }
00163    
00164    if (_displayfunction & LCD_8BITMODE)
00165    {
00166       writeNbits(value, 8); 
00167    } 
00168    else 
00169    {
00170       writeNbits ( value >> 4, 4 );
00171       writeNbits ( value, 4 );
00172    }
00173    waitUsec ( EXEC_TIME ); // wait for the command to execute by the LCD
00174 }
