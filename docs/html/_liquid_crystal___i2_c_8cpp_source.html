00176    // No need to use the delay routines since the time taken to write takes

00183 //
00184 // write4bits
00185 void LiquidCrystal_I2C::write4bits ( uint8_t value, uint8_t mode ) 
00186 {
00187    uint8_t pinMapValue = 0;
00188    
00189    // Map the value to LCD pin mapping
00190    // --------------------------------
00191    for ( uint8_t i = 0; i < 4; i++ )
00192    {
00193       if ( ( value & 0x1 ) == 1 )
00194       {
00195          pinMapValue |= _data_pins[i];
00196       }
00197       value = ( value >> 1 );
00198    }
00199    
00200    // Is it a command or data
00201    // -----------------------
00202    if ( mode == DATA )
00203    {
00204       mode = _Rs;
00205    }
00206    
00207    pinMapValue |= mode | _backlightStsMask;
00208    _i2cio.write ( pinMapValue );
00209    pulseEnable ( pinMapValue );
00210 }
